\begin{frame}{Motivation: Need for Automated Academic Presentation Videos}
\begin{columns}
\begin{column}{0.55\textwidth}
\begin{itemize}
  \item Academic presentation videos critical for research dissemination.
  \item Manual production is highly \alertterm{labor-intensive}, requiring hours for just a few minutes.
  \item Unique challenges:
  \begin{itemize}
    \item Long-context multi-modal input from dense scientific papers.
    \item Multi-channel alignment: slides, subtitles, speech, talking-head, cursor.
    \item No existing benchmarks or tailored metrics to evaluate quality.
  \end{itemize}
  \item \alertterm{Key questions:}
  \begin{itemize}
    \item What defines a \alertterm{good academic presentation video}?
    \item How to \alertterm{automatically generate} such videos from papers?
  \end{itemize}
\end{itemize}
\end{column}

\includegraphics[width=1\textwidth]{figure/teaser.pdf}

\begin{frame}{Related Work: Landscape and Challenges}
\begin{itemize}
  \item \textbf{Natural Video Generation:}
  \begin{itemize}
    \item Diffusion models generate short videos from text~\cite{sd_video,wan}.
    \item Long and multi-shot videos remain challenging~\cite{deepmind2025veo3}.
  \end{itemize}
  \item \textbf{Multi-agent frameworks for videos:}
  \begin{itemize}
    \item MovieAgent generates movies via role decomposition~\cite{wu2025automated}.
    \item PresentAgent combines slide and speech but lacks personalization and academic slide quality~\cite{shi2025presentagent}.
  \end{itemize}
  \item \textbf{AI for Research:}
  \begin{itemize}
    \item Automatic generation of scientific posters~\cite{pang2025paper2poster} and slides~\cite{zheng2025pptagent}.
    \item Few efforts on \alertterm{academic presentation video synthesis}.
  \end{itemize}
  \item \alertterm{Our work fills this gap}, focusing on long, multimodal, personalized academic video generation with comprehensive evaluation.
\end{itemize}
\end{frame}

\begin{frame}{Method Overview: The PaperTalker Agent}
\begin{figure}
    \centering
    \includegraphics[width=0.75\textwidth]{figure/method.pdf}
    \caption{Pipeline of \alertterm{PaperTalker}: Modular multi-agent approach integrating slide generation, subtitle/cursor creation, and talking-head synthesis.}
    \label{fig:method}
\end{figure}
\vspace{-0.8cm}
\begin{block}{Key Modules}
\begin{itemize}
  \item \alertterm{Slide Builder}: Generates \LaTeX~Beamer slides, refines layouts via compilation feedback.
  \item \alertterm{Subtitle \& Cursor Builder}: Produces aligned subtitles with spatially grounded cursor trajectories.
  \item \alertterm{Talker Builder}: Synthesizes personalized speech and renders talking-head videos.
  \item Supports \alertterm{slide-wise parallelization} for efficiency.
\end{itemize}
\end{block}
\end{frame}

% Method 1: Slide Builder with layout refinement
% corresponds to source: Slide Builder subsection in method
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Method: Slide Builder and Layout Refinement}
\begin{itemize}
  \item \alertterm{Slides are generated directly} from the paper’s \LaTeX~source using Beamer for professional academic layouts.
  \item Compilation feedback (errors, warnings) used to iteratively fix code ensuring correctness.
  \item \alertterm{Tree Search Visual Choice:}
  \begin{itemize}
    \item Rule-based generation of layout variants by adjusting font and figure scales.
    \item Use vision-language model to \alertterm{select the best candidate visually}.
    \item Solves overflow and layout issues efficiently beyond parameter tuning by LLMs.
  \end{itemize}
\end{itemize}

    \centering
    \includegraphics[width=0.5\textwidth]{figure/tree_search.pdf}
    \caption{Tree Search Visual Choice for slide layout refinement.}

\end{frame}

% Method 2: Subtitle and Cursor Builder
% corresponds to source: Subtitle Builder and Cursor Builder subsections
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Method: Subtitle and Cursor Builder}
\begin{itemize}
  \item Subtitles are generated for each slide with a vision-language model.
  \item \alertterm{Visual focus prompts} highlight key content for each subtitle sentence.
  \item Cursor trajectory is \alertterm{spatially aligned} per sentence using GUI-grounding models.
  \item \alertterm{Temporal alignment} with narration timing achieved using WhisperX.
\end{itemize}
\begin{block}{Benefit}
\begin{itemize}
  \item \alertterm{Enhanced audience guidance} by highlighting the precise focus of the narration on the slides.
  \item Improves accessibility and content grounding.
\end{itemize}
\end{block}
\end{frame}

% Method 3: Talker Builder and Parallelization
% corresponds to source: Talker Builder subsection
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Method: Talker Builder and Parallelization}
\begin{itemize}
  \item Personalize speech synthesis using author voice sample with F5-TTS at slide granularity.
  \item Talking-head video generated based on author portrait using FantasyTalking and Hallo2.
  \item \alertterm{Slide-wise parallel generation} for significant acceleration (6$\times$ speedup).
  \item You save hours of generation time while preserving video quality and engagement.
\end{itemize}

\begin{frame}{Experimental Settings}
\begin{itemize}
  \item \alertterm{Baselines:}
  \begin{itemize}
    \item End-to-end video generation models (e.g., Wan2.2, Veo3).
    \item Multi-agent frameworks (e.g., PresentAgent).
    \item Human-created presentations as upper bound.
  \end{itemize}
  \item \alertterm{Dataset:} Paper2Video benchmark with 101 paper-video pairs from recent AI conferences.
  \item Hardware: 8$\times$ NVIDIA RTX A6000 GPUs.
  \item Evaluation Models:
  \begin{itemize}
    \item Vision-language model: GPT-4.1.
    \item Video language model: Gemini-2.5-Flash.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Limitations and Future Research Directions}
\begin{itemize}
  \item \alertterm{Current Limitations:}
  \begin{itemize}
    \item Difficulty in parsing long scientific documents with complex equations end-to-end.
    \item Some layouts and slide designs still need refinement.
    \item Talking-head synthesis quality and naturalness can be improved further.
  \end{itemize}
  \item \alertterm{Future Directions:}
  \begin{itemize}
    \item Real-time interactive presentation video agents enabling Q\&A and dynamic tutorials.
    \item Enhanced semantic understanding of longer, complex papers.
    \item Support for multilingual presentations and accessibility features.
    \item Closed-loop evaluation with human and AI-in-the-loop feedback.
    \item Full open-sourcing for community collaboration.
  \end{itemize}
\end{itemize}
